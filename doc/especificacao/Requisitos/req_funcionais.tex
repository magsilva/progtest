\section{Requisitos Funcionais}

 Os requisitos funcionais do ambiente são:\\


\textbf{Gerais:}

\begin{enumerate}
    \item  Cadastro de usuários (alunos e professores) (inserir, alterar, excluir e visualizar)
            \begin{enumerate}[{1.}1]
                \item Inserir
                \item Alterar
                \item Excluir
                \item Visualizar
            \end{enumerate}
    \item Autenticação de usuários
    \item Cadastro de disciplinas
            \begin{enumerate}[{3.}1]
                \item Inserir
                \item Alterar
                \item Excluir
                \item Visualizar
            \end{enumerate}
\end{enumerate}



\textbf{ Professor:}

 \begin{enumerate}
    \item Cadastrar turmas
        \begin{enumerate}[{1.}1]
            \item Inserir
            \item Alterar
            \item Excluir
            \item Visualizar
        \end{enumerate}
    \item Adicionar alunos às turmas
    \item Cadastrar trabalhos
        \begin{enumerate}[{3.}1]
            \item Inserir
            \item Alterar
            \item Excluir
            \item Visualizar
            \item Inserir oráculos (programas e casos de teste)
        \end{enumerate}
    \item Visualizar o histórico de cada aluno
        \begin{enumerate}[{4.}1]
            \item Quais trabalhos foram entregues
            \item Quais foram os resultados obtidos
        \end{enumerate}
 \end{enumerate}



\textbf{Alunos:}

\begin{enumerate}
    \item Visualizar turma
    \item Visualizar trabalhos (descrição, datas):
        \begin{enumerate}[{2.}1]
            \item Realizados (Mostrar resultados da avaliação)
            \item Em Andamento (Mostrar datas finais, permitir
      submissão)
        \end{enumerate}
    \item Submeter programas (Permitir submeter qualquer tipo de arquivo
 ??)
    \item Submeter casos de teste
    \item Visualizar resultados da avaliação
        \begin{enumerate}
            \item Por critérios
            \item Aceito ou não
            \item Nota atribuída ao trabalho
        \end{enumerate}
\end{enumerate}


\textbf{Submissão:}

\begin{enumerate}
    \item Permitir o upload do arquivo do programa
    \item Permitir o upload do arquivo com os casos de teste
    \item Receber somente arquivos do tipo class
\end{enumerate}

\textbf{Avaliação:}

\begin{enumerate}
    \item Integrar com a ferramenta de teste Jabuti
    \item Criar uma fórmula para dar a nota do aluno:
        \begin{enumerate}[{2.}1]
            \item Baseada nos critérios de teste de software cobertos pela ferramenta
     Jabuti
            \item De acordo com o estágio que o aluno está no curso
                \begin{enumerate}[{2.2.}1]
                    \item Pesos devem ser atribuídos a cada critério
           coberto pela ferramenta Jabuti
                    \item Levar em consideração os caminhos
           não-executáveis???
                    \item Mostrar aos alunos a implementação do
           professor???
                    \item Criar um conjunto com todos os casos de teste
           desenvolvidos por todos os alunos e usar esse conjunto
           na avaliação dos programas ???
                \end{enumerate}
        \end{enumerate}
\end{enumerate}

\subsection{Gerais}\label{subsec:reqgerais}

\subsubsection{Cadastro de usuários}

\begin{itemize}

    \item Requisito Funcional 1: Inserir usuários

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        inserir usuários no sistema.
        \item Entrada $\rightarrow$ Deve serem fornecidos o nome,
        o número de matrícula, tipo de usuário (professor ou
        aluno), email, login e senha.
        \item Processamento $\rightarrow$ Verificação se o número de matrícula está correto.
        \item Saída $\rightarrow$ Usuário cadastrado.
    \end{itemize}

\end{itemize}

\begin{itemize}

    \item Requisito Funcional 2: Alterar usuários

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        alterar usuários no sistema.
    \end{itemize}

\end{itemize}

\begin{itemize}

    \item Requisito Funcional 3: Excluir usuários

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        excluir usuários do sistema.
    \end{itemize}

\end{itemize}

\begin{itemize}

    \item Requisito Funcional 4: Visualizar usuários

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        visualizar usuários no sistema.
     \end{itemize}

\end{itemize}



\begin{itemize}

    \item Requisito Funcional 5: Autenticação de usuários

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de fazer a autenticação de usuários.
        \item Entrada $\rightarrow$ Deve serem fornecidos o login e a senha.
        \item Processamento $\rightarrow$ Verificação se o login e a senha estão corretos.
        \item Saída $\rightarrow$ Usuário autenticado no sistema ou uma mensagem de erro indicando que o login
        ou a senha são inválidos.
    \end{itemize}

\end{itemize}

    \item Requisito Funcional 9: Visualizar disciplinas

\subsection{Visão Professor}\label{subsec:reqprofessor}

\begin{itemize}

    \item Requisito Funcional 10: Criar turma

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        permitir o criação de turmas. Somente o usuário do
        tipo professor tem o poder de realizar essa função.
        \item Entrada $\rightarrow$ Devem ser fornecidos somente o
        o período em a disciplina é realizada.
    \end{itemize}

\end{itemize}

\begin{itemize}

    \item Requisito Funcional 11: Alterar turma

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve permitir que os professores
        alterem os dados das turmas existentes.
        \item Entrada $\rightarrow$ Os dados anteriores da turma.

    \end{itemize}

\end{itemize}

\begin{itemize}

    \item Requisito Funcional 12: Excluir turma

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve permitir que os professores
        excluem turmas existentes.
    \end{itemize}

\end{itemize}

\begin{itemize}

    \item Requisito Funcional 13: Visualizar turma

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve permitir
        visualização das turmas para todos os usuários.
    \end{itemize}

\end{itemize}


\begin{itemize}

    \item Requisito Funcional 14: Adicionar alunos às turmas

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        adicionar alunos a uma determina turma. Somente o
        usuário tipo professor tem o poder de realizar essa
        tarefa.
        \item Entrada $\rightarrow$ Devem ser fornecidos o nome e
        número de matrícula do aluno.
        \item Processamento $\rightarrow$ O sistema deve verificar
        se o aluno já está cadastrado no sistema. O sistema deve
        verificar se a quantidade de alunos da turma não foi
        ultrapassada.
        \item Saída $\rightarrow$ Mensagem indicando sucesso ou
        mensagem indicando erro.
    \end{itemize}

\end{itemize}

\begin{itemize}

    \item Requisito Funcional 15: Inserir trabalhos

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        permitir a inserção de trabalhos. Somente o
        usuário tipo professor tem o poder de realizar essa
        tarefa.
        \item Entrada $\rightarrow$ Devem ser fornecidos o título
        e a descrição do trabalho, data de início da submissão, data final
        da submissão e os arquivos contendo o oráculo(tanto programa e casos de teste).
        \item Processamento $\rightarrow$ O sistema deve verificar
        se os arquivos são realmente do programa e dos casos de teste fornecidos como
        oráculo.
        \item Saída $\rightarrow$ Mensagem indicando sucesso ou
        mensagem indicando erro.
    \end{itemize}

\end{itemize}


\begin{itemize}

    \item Requisito Funcional 16: Alterar trabalhos

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        permitir a alteração de trabalhos existentes. Somente o
        usuário tipo professor tem o poder de realizar essa
        tarefa.
        \item Entrada $\rightarrow$  Os dados anteriores do trabalho existente.
    \end{itemize}

\end{itemize}



\begin{itemize}

    \item Requisito Funcional 17: Excluir trabalhos

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        excluir trabalhos existentes. Somente o
        usuário tipo professor tem o poder de realizar essa
        tarefa.
    \end{itemize}

\end{itemize}


\begin{itemize}

    \item Requisito Funcional 18: Visualizar trabalhos

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve permitir que todos os usuários
        visualizem os trabalhos existentes.
    \end{itemize}

\end{itemize}


\begin{itemize}

    \item Requisito Funcional 19: Inserir alunos

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        permitir a inserção de dados dos alunos.
        \item Entrada $\rightarrow$ Devem ser fornecidos o nome,
        o número de matrícula, login, senha e email.
        \item Processamento $\rightarrow$ Verificação se o número de matrícula está correto.
        \item Saída $\rightarrow$ Usuário cadastrado.
    \end{itemize}

\end{itemize}


\begin{itemize}

    \item Requisito Funcional 20: Alterar alunos

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        permitir a alteração de dados dos alunos.
        \item Entrada $\rightarrow$ Os dados anteriores do aluno existente.
    \end{itemize}

\end{itemize}


\begin{itemize}

    \item Requisito Funcional 21: Excluir alunos

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        permitir a exclusão de dados dos alunos.
    \end{itemize}

\end{itemize}


\begin{itemize}

    \item Requisito Funcional 22: Guardar histórico dos alunos

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        guardar o histórico das submissões e avaliações dos
        trabalhos realizados de todos os alunos.
    \end{itemize}

\end{itemize}


\begin{itemize}

    \item Requisito Funcional 23: Visualizar alunos

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve permitir que todos os usuários
        possam visualizar os dados dos alunos.
    \end{itemize}

\end{itemize}



\subsection{Visão Aluno}\label{subsec:reqalunos}


\begin{itemize}

    \item Requisito Funcional 24: Submeter programas

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve permitir que
        os alunos submetam programas ao sistema.
        \item Entrada $\rightarrow$ O arquivo contendo o programa desenvolvido pelo aluno.
        \item Processamento $\rightarrow$ Deve ser verificado se o
        arquivo submetido é realmente um arquivo contendo um
        programa.
        \item Saída $\rightarrow$ Mensagem indicando sucesso ou
        erro na submissão.
     \end{itemize}

\end{itemize}


\begin{itemize}

    \item Requisito Funcional 25: Submeter casos de teste

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve permitir que
        os alunos submetam casos de teste ao sistema.
        \item Entrada $\rightarrow$ O arquivo contendo os casos de teste  desenvolvidos pelo aluno.
        \item Processamento $\rightarrow$ Deve ser verificado se o
        arquivo submetido é realmente um arquivo contendo casos de
        teste.
        \item Saída $\rightarrow$ Mensagem indicando sucesso ou
        erro na submissão.
     \end{itemize}

\end{itemize}

\begin{itemize}

    \item Requisito Funcional 26: Visualizar resultados da
    avaliação dos programas e casos de teste

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve permitir que
        todos os usuários visualizem o resultado da avaliação
        dos trabalhos (por critérios, se foi aceito ou não, nota atribuída)
        dos alunos.
    \end{itemize}

\end{itemize}



\subsection{Submissão}\label{subsec:submissao}

\begin{itemize}

    \item Requisito Funcional 27: Aceitar submissão de programas e
    casos de teste

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        aceitar a submissão de programas e casos de teste.
        \item Entrada $\rightarrow$ Os arquivos desenvolvidos
        pelos alunos.
        \item Processamento $\rightarrow$ O sistema deve fazer o
        upload dos arquivos e verificar se os arquivos submetidos
        estão corretos.
        \item Saída $\rightarrow$ Mensagem indicando sucesso ou erro.
    \end{itemize}

\end{itemize}


\subsection{Avaliação - Detalhar mais}\label{subsec:submissao}

\begin{itemize}

    \item Requisito Funcional 28: Avaliar programas e casos de
    teste

    \begin{itemize}
        \item Descrição $\rightarrow$ O sistema deve ser capaz de
        avaliar os programas e casos de teste submetidos.
        \item Entrada $\rightarrow$ Os arquivos submetidos pelos
        alunos.
        \item Processamento $\rightarrow$ O sistema deve analisar
        os arquivos e avaliar os programas e casos de teste de
        acordo com critérios definidos.
        \item Saída $\rightarrow$ Resultados da avaliação.
    \end{itemize}

\end{itemize}
